#include <iostream>
#include <fstream>
#include <vector>
#include <string>
using namespace std;

class Task {
    string text; bool done;
 public:
    Task(string t=""): text(t), done(false) {}
    string getText(){
           return text; }
    bool isDone(){
           return done; }
    void mark(){ done= true; }
};

class Manager {
    vector<Task> tasks;
    string file="tasks.txt";
    void save(){
        ofstream f(file);
        if(!f){ cout<<"Error file \n";
           return; }
        for(auto &t: tasks){
            f<<t.getText()<<"|"<<(t.isDone()?1:0)<<"\n";
        }
    }
    void load(){
        ifstream f(file);
        if(!f) return;
        tasks.clear();
        string line;
        while(getline(f,line)){
            int pos=line.find('|');
            if(pos==string::npos) 
            continue;
            string t=line.substr(0,pos);
            string s=line.substr(pos+1);
            Task tmp(t);
            if(s=="1") tmp.mark();
            tasks.push_back(tmp);
        }
    }
public:
    Manager(){ load(); }
    void add(string t){
        if(t.empty()){ cout<<"Empty task: \n"; 
         return;}
        tasks.push_back(Task(t));
        save();
        cout<<"Added :\n";
    }
    void complete(int n){
        if(n<0 || n>=tasks.size()){ cout<<"Bad index!\n"; return; }
        if(tasks[n].isDone()) cout<<"Already done!\n";
        else{ tasks[n].mark(); save(); cout<<"Ok, completed\n"; }
    }
    void show(){
        if(tasks.empty()){ cout<<"(No tasks)\n"; return;}
        for(int i=0;i<tasks.size();i++)
            cout<<i+1<<". "<<tasks[i].getText()
                <<" ["<<(tasks[i].isDone()?"Done":"Pending")<<"]\n";
    }
};

int main(){
    
    Manager m; string in;
    cout<<"-- Task Manager --\n";
     while(true){
        cout<<"\n1.Add 2.Complete 3.Show 4.Exit\n> ";
        getline(cin,in);
        if(in=="1"){
            cout<<"Enter: "; getline(cin,in);
            if(in.size()) m.add(in);
        } else if(in=="2"){
            m.show(); cout<<"Which #: ";
            getline(cin,in);
            try{ int n=stoi(in)-1; m.complete(n); }
            catch(...){ cout<<"Not number!\n"; }
        } else if(in=="3") m.show();
        else if(in=="4"){ cout<<"Bye\n"; break; }
        else cout<<"?? invalid\n";
    }
}
